(def pou (atom {:params (js/klipse.utils.url-parameters)
                    :editors {:main {:p (aget js/klipse-editors 0)
                                     :o (aget js/klipse-results 0)
                                     :u js/klipse-container}}}))
    (let [{:keys [params editors]} @pou] (if-let [p (params :p)] (.setValue (-> editors :main :p) (js/atob p)) @pou))
